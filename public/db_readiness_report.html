WITH tables_check AS ( SELECT COUNT(*) AS total, COUNT(t.table_name) AS found FROM (VALUES ('attribute_definitions'),('clubs'),('coaches'),('fixtures'), ('leagues'),('market_value_history'),('match_events'),('match_lineups'), ('notification_recipients'),('notifications'),('player_attribute_history'), ('player_attributes'),('player_development_projections'), ('player_fitness_snapshots'),('player_guardians'),('player_hidden_attributes'), ('player_injury_risk_profiles'),('player_market_values'),('player_match_stats'), ('player_morale_snapshots'),('player_ownership_claims'), ('player_passport_score_history'),('player_position_familiarity'), ('player_similarities'),('player_training_attendance'), ('player_training_performance'),('players'),('profiles'),('seasons'), ('standings'),('training_sessions'),('user_follows') ) AS req(table_name) LEFT JOIN information_schema.tables t ON t.table_name = req.table_name AND t.table_schema = 'public' AND t.table_type = 'BASE TABLE' ), views_check AS ( SELECT COUNT(*) AS total, COUNT(v.viewname) AS found FROM (VALUES ('v_development_leaderboard'),('v_pipeline_summary'), ('v_player_development_passport'),('v_player_dna_public'), ('v_player_market_values_public'),('v_player_passport_full'), ('v_reputation_leaderboard'),('v_scout_reports_public'), ('v_squad_development_report'),('v_wonderkid_radar') ) AS req(view_name) LEFT JOIN pg_views v ON v.viewname = req.view_name AND v.schemaname = 'public' ), mv_check AS ( SELECT COUNT(*) AS total, COUNT(m.matviewname) AS found FROM (VALUES ('mv_club_dna'),('mv_league_standings'), ('mv_player_passport_scores'),('mv_top_scorers') ) AS req(mv_name) LEFT JOIN pg_matviews m ON m.matviewname = req.mv_name AND m.schemaname = 'public' ), rpc_check AS ( SELECT COUNT(*) AS total, COUNT(p.proname) AS found FROM (VALUES ('compute_weekly_training_score'),('get_my_player_id'), ('run_post_match_pipeline'),('search_players') ) AS req(fn_name) LEFT JOIN pg_proc p ON p.proname = req.fn_name AND p.pronamespace = 'public'::regnamespace ) SELECT tc.found || '/' || tc.total AS tables, vc.found || '/' || vc.total AS views, mc.found || '/' || mc.total AS matviews, rc.found || '/' || rc.total AS rpcs, (tc.found+vc.found+mc.found+rc.found) || '/' || (tc.total+vc.total+mc.total+rc.total) AS total_objects, ROUND( (tc.found+vc.found+mc.found+rc.found)::numeric / (tc.total+vc.total+mc.total+rc.total)::numeric * 100, 1 ) AS completeness_pct FROM tables_check tc, views_check vc, mv_check mc, rpc_check rc;
